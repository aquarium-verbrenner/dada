public class ControlFlowTestDrive {
  
  public static void main(String[] args) {
    Scanner inputScanner = new Scanner(System.in);

    String name = inputScanner.nextLine();
    for (int i = 0; i < 5; i++) {

    }

     inputScanner.close();
  }
